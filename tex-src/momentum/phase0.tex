%=== Phase-0 : モメンタム係数 γ_t ベースライン ================================
\section*{Phase 0：モメンタム係数 $\gamma_t$ ベースライン}\nopagebreak[4]
%────────────────────────────────────
\begin{flushleft}
\[
  \boxed{\gamma_t = 0}
\]
\end{flushleft}

\subsection*{変数のポイント}
\begin{flushleft}
\begin{itemize}
  \item データ欠損・学習初期のフォールバックとして **常に \(\gamma_t=0\)** を使用。  
        始値と終値のシフトを一切行わない。
\end{itemize}
\end{flushleft}

\subsection*{変数メモ}
\begin{flushleft}
本フェーズは学習初期・データ欠損時のフォールバックとして使用し、  
当日寄り付きと引けの偏位を考慮しない設定（$\gamma_t=0$）を採用する。
\end{flushleft}
