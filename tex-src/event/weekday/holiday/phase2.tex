\subsection*{ステップ・目的}
\begin{flushleft}
\begin{enumerate}
  \item \textbf{入力係数作成}\;
        \(\beta_{\text{weekday},i,t}^{\ast}
          =\beta_{\text{weekday},i,t}^{(2)}\,\beta_{\text{holiday},t}\)
  \item \textbf{平滑定数}\;
        \(\lambda_{\text{wd}}=0.90\)
  \item \textbf{EWMA 更新}\;
        \[
          \hat\beta_{\text{weekday},i,t}
            =\lambda_{\text{wd}}\,
             \hat\beta_{\text{weekday},i,t-1}
            +(1-\lambda_{\text{wd}})\,
             \beta_{\text{weekday},i,t}^{\ast}
        \]
\end{enumerate}
\end{flushleft}
